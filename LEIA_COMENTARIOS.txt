EXEMPLO: FIFO

/**
 * Algoritmo FIFO (First In First Out)
 * 
 * CONCEITO:
 * O FIFO substitui a página que está há mais tempo na memória.
 * Funciona como uma fila: primeira a entrar é primeira a sair.
 * 
 * LÓGICA:
 * 1. Mantém array representando fila de páginas
 * 2. Se página não está na memória (FALTA), incrementa contador
 * 3. Se há espaço livre, adiciona página ao final
 * 4. Se está cheio, remove página mais antiga e adiciona nova
 * 
 * COMPLEXIDADE: O(n) onde n é comprimento da sequência
 */

// DEPOIS vem o código bem comentado linha por linha:

for (let page of seq) {
    // Cria snapshot do estado atual
    let state = {
        page: page,                    // Página sendo acessada
        memory: [...memory],           // Cópia da memória
        isFault: !memory.includes(page) // Hit ou Falta?
    };

    // Se página NÃO está na memória (FALTA)
    if (!memory.includes(page)) {
        faults++;  // Incrementa contador
        
        if (memory.length < quadros) {
            // Espaço vazio: apenas adiciona
            memory.push(page);
        } else {
            // Está cheio: remove + antiga, adiciona nova
            memory.shift(); // Remove primeiro (mais antigo)
            memory.push(page); // Adiciona nova
        }
    }
    // Se includes() = true, é um HIT, não faz nada

    history.push(state);  // Guarda estado para referência
}
